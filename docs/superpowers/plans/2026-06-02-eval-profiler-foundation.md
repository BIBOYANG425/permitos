```python
# research_aiq/research_aiq/eval_report.py
"""Post-process a `nat eval` output dir into derived headline metrics + a committed
scorecard. Pure functions (this section) are unit-tested over a fixture; the output-dir
ADAPTER (Task 4) reads nat's real files. Dollar cost is derived from profiler token
counts x an in-repo per-model price map (explicit + updatable, not a live API call).
The profiler captures the ORCHESTRATION LLM only; Modal researcher LLM cost runs on
Modal and is out of nat's profiler scope (documented in the scorecard)."""
from __future__ import annotations

# $ per 1M tokens (input, output). Update as pricing changes.
MODEL_PRICING: dict[str, dict[str, float]] = {
    "gpt-5.5": {"input": 5.0, "output": 30.0},
    "gpt-5.2": {"input": 1.75, "output": 14.0},
    "gpt-4o-mini": {"input": 0.15, "output": 0.60},
    "o4-mini": {"input": 1.1, "output": 4.4},
}


def cost_from_usage(usage: dict[str, dict[str, int]], pricing: dict) -> float:
    """Sum $ over per-model token usage. Unknown models contribute 0 (caller reports them)."""
    total = 0.0
    for model, toks in usage.items():
        price = pricing.get(model)
        if not price:
            continue
        total += toks.get("input_tokens", 0) / 1_000_000 * price["input"]
        total += toks.get("output_tokens", 0) / 1_000_000 * price["output"]
    return total


def percentile(values: list[float], p: float) -> float | None:
    """Nearest-rank percentile; None for empty input (fail-soft)."""
    if not values:
        return None
    s = sorted(values)
    k = max(1, math.ceil(p / 100 * len(s)))
    return s[k - 1]


def derive_run_metrics(run: dict, pricing: dict) -> dict:
    """Derived per-run headline metrics from extracted primitives."""
    cost = cost_from_usage(run.get("usage", {}), pricing)
    n_det = run.get("n_determinations", 0) or 0
    lat = run.get("researcher_latencies_ms", []) or []
    return {
        "cost_usd": cost,
        "cost_per_determination_usd": (cost / n_det) if n_det else None,
        "n_researchers": len(lat),
        "modal_latency_p50_ms": percentile(lat, 50),
        "modal_latency_p95_ms": percentile(lat, 95),
    }


import math  # noqa: E402  (kept after the docstring for readability)
```

```python
def load_eval_output(output_dir: str | Path) -> dict:
    """Read nat's eval output dir into {'evaluators': {name: aggregate_score},
    'runs': [run dict for derive_run_metrics], 'model': str}.

    d = Path(output_dir)

        f = d / f"{name}_output.json"   # adjust to the fixture's real filename
        if f.exists():
            obj = json.loads(f.read_text())
            evaluators[name] = obj.get("average_score", obj.get("score"))
    # Profiler report -> per-run usage + latencies; ATIF outputs -> n_determinations.
    runs: list[dict] = _runs_from_profiler_and_atif(d)   # implement per Task-2 format
    model = runs[0]["model"] if runs else None

@dataclass
class Scorecard:
    evaluators_primary: dict = field(default_factory=dict)
    evaluators_directional: dict = field(default_factory=dict)

    sc.evaluators_primary = {k: data["evaluators"].get(k) for k in PRIMARY}
    sc.evaluators_directional = {k: data["evaluators"].get(k) for k in DIRECTIONAL}
    sc.run_metrics = [derive_run_metrics(r, pricing) for r in data["runs"]]
    costs = [m["cost_usd"] for m in sc.run_metrics]
    sc.aggregate = {
        "n_runs": len(sc.run_metrics),
        "total_cost_usd": sum(costs),
        "mean_cost_per_run_usd": (sum(costs) / len(costs)) if costs else None,
    }
    seen = {m for r in data["runs"] for m in r.get("usage", {})}
    sc.unpriced_models = sorted(m for m in seen if m not in pricing)
    return sc


def render_scorecard_md(sc: Scorecard) -> str:
    lines = ["# research_aiq eval scorecard", ""]
    lines.append("## Primary metrics (rigorous)")
    for k, v in sc.evaluators_primary.items():
        lines.append(f"- **{k}**: {v}")
    lines.append("")
    lines.append("## Directional metric (not a rigorous benchmark)")
    for k, v in sc.evaluators_directional.items():
        lines.append(f"- {k} (directional): {v}")
    lines.append("")
    lines.append("## Cost / latency (derived; orchestration LLM only — Modal researcher LLM cost is separate)")
    a = sc.aggregate
    lines.append(f"- runs: {a['n_runs']}  total ${a['total_cost_usd']:.4f}  mean cost/run ${a['mean_cost_per_run_usd']}")
    cpd = [m["cost_per_determination_usd"] for m in sc.run_metrics if m["cost_per_determination_usd"]]
    if cpd:
        lines.append(f"- cost_per_determination (mean): ${sum(cpd)/len(cpd):.4f}")
    p50 = [m["modal_latency_p50_ms"] for m in sc.run_metrics if m["modal_latency_p50_ms"]]
    if p50:
        lines.append(f"- Modal researcher latency p50 (mean of per-run p50): {sum(p50)/len(p50):.0f} ms")
    if sc.unpriced_models:
        lines.append(f"- NOTE unpriced models (cost excluded): {', '.join(sc.unpriced_models)}")
    return "\n".join(lines) + "\n"


def main(output_dir: str, out_md: str, out_json: str) -> None:
    data = load_eval_output(output_dir)
    sc = build_scorecard(data)
    Path(out_md).write_text(render_scorecard_md(sc))
    Path(out_json).write_text(json.dumps({
        "evaluators_primary": sc.evaluators_primary,
        "evaluators_directional": sc.evaluators_directional,
        "aggregate": sc.aggregate,
        "run_metrics": sc.run_metrics,
        "unpriced_models": sc.unpriced_models,
    }, indent=2))


if __name__ == "__main__":
```
